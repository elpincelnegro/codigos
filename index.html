def prod(matriz1,matriz2):
    for i in range(len(matriz1)):
        final = []
        for j in range(len(matriz1)):
            p = 0
            for k in range(len(matriz1)):
                p += matriz1[i][k]*matriz2[k][j]
            final = final+[p]
        print(final)
